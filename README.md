From Scratch Project Setup:
// package.json & README
$ npm init
$ touch README.md
$ touch git.ignore // add 'node_modules'
$ git init
$ mkdir src dist
$ touch src/index.html

// Gulp
$ npm install -g gulp
$ npm install --save gulp@3.9.0 gulp-connect@2.2.0 gulp-open@1.0.0
$ touch gulpfile.js // dig into file for configuration
